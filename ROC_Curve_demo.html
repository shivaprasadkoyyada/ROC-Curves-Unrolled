# define lists tpr,fpr,tnr,ppr
def ratios(Actualvalues,predictions_threshold):
    ### BEGIN HIDDEN SOLUTION
    tpr=[]
    fpr=[]
    tnr=[]
    ppr=[]
# Write a for loop to iterate with different threshold values; 
# Note: to come out of divided by zero error, you have to think a littile bit.

    for j in range(0,9):
        tp=0
        fp=0
        tn=0
        fn=0
        for i in range(0,15):
            if(Actualvalues[i]==1 and predictions_threshold[j][i]==1):
                tp=tp+1
            if(Actualvalues[i]==1 and predictions_threshold[j][i]==0):
                fn=fn+1
            if(Actualvalues[i]==0 and predictions_threshold[j][i]==1):
                fp=fp+1
            if(Actualvalues[i]==0 and predictions_threshold[j][i]==0):
                tn=tn+1 
        if(tp==0):
            tpr.append(0)
            ppr.append(0)
        else:
            tpr.append(tp/(tp+fn))
            ppr.append(tp/(tp+fp))
    
        if(tn==0):
            tnr.append(0)
            fpr.append(1)
        else:
            tnr.append(tn/(tn+fp))
            fpr.append(fp/(fp+tn))
    ### END HIDDEN SOLUTION
    return tpr,fpr,tnr,ppr
